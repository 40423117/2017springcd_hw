###加減乘除
<pre class="brush:html;auto-links:false;toolbar:false" contenteditable="false">
def addend (a,b): #加運算
      return a+b
def minus (a,b): #減運算
      return a-b
def tines (a,b):    #乘運算
      return a*b
def divided (a,b):  #除運算
      return a/b
</pre>
